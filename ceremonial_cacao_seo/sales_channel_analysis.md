      function getPriceAmount(text) {
        /*
         INFO:
         - extracts a string from a pattern: [number][separator][number][separator]...
           [separator] - any symbol but a digit, including whitespaces
         */
        const pattern = /\d+(?:[^\d]\d+)+(?!\d)/g;
        const matches = text.match(pattern);
        const results = [];

        // if there are no matches
        if (!matches?.length) {
          const amount = text.replace(/[^\d]/g, '');

          if (!amount) return results;

          results.push(amount);

          return results;
        }

        matches.forEach(str => {
          const sepIndexes = [];

          for (let i = 0; i < str.length; i++) {
            // find separator index and push to array
            if (!/\d/.test(str[i])) {
              sepIndexes.push(i);
            }
          }
          if (sepIndexes.length === 0) return;
          // assume the last separator is the decimal
          const decIndex = sepIndexes[sepIndexes.length - 1];

          // cut the substr to the separator with replacing non-digit chars
          const intSubstr = str.slice(0, decIndex).replace(/[^\d]/g, '');
          // cut the substr after the separator with replacing non-digit chars
          const decSubstr = str.slice(decIndex + 1).replace(/[^\d]/g, '');

          if (!intSubstr) return;

          const amount = decSubstr ? `${intSubstr}.${decSubstr}` : intSubstr;
          // to get floating point number
          const number = parseFloat(amount);

          if (!isNaN(number)) {
            results.push(String(number));
          }
        });

        return results;
      }

      function loadScript(src, attributes) {
        return new Promise((resolve, reject) => {
          const scriptTag = document.createElement('script');

          if (attributes) {
            Object.keys(attributes)
                    .forEach(key => {
                      scriptTag.setAttribute(key, attributes[key]);
                    });
          }

          scriptTag.type = 'text/javascript';
          scriptTag.src = src;

          scriptTag.onload = () => resolve();
          scriptTag.onerror = event => {
            reject(event);
          };

          document.head.appendChild(scriptTag);
        });
      }

      function transformConfig(config) {
        let transformedConfig = {};

        const keys = Object.keys(config);

          const value = config[key];

          if (
                  value === undefined ||
                  value === null ||
                  value === '' ||
                  (Array.isArray(value) && value.length === 0)
          ) {
            return;
          }

          transformedConfig = {
            ...transformedConfig,
            [key]: Array.isArray(value) ? value.join(',') : value,
          };
        });

        return transformedConfig;

## Executive analysis: what’s working and what’s not

### Overall

- **4 of 5** competitor sites loaded and had a usable shop → product → cart flow. **Ora Cacao** did not load within 35s (timeout), so its sales channel could not be assessed.
- **Platform:** 3 use **Shopify**, 1 uses **BigCommerce**. E‑commerce stack is consistent and standard.
- **Price visibility:** All 4 assessed sites use layouts or selectors where “price visible” was detected as false by the script (price exists in the DOM but is not in a single clear element). So **price clarity on product pages** is a common weakness from an automated snapshot; real users may still see prices.
- **Checkout path:** Only **Bar and Cocoa** and **Ceremonial Cacao** showed a clear path to checkout (cart page with visible checkout CTA). **Four Visions** and **Ma Cacao** had cart pages reachable but the checkout CTA was not obvious to the script (e.g. different selector or layout).

### By brand

1. **Ceremonial Cacao (Shopify)**  
   - **Working:** Shop discoverable, strong product count (39), add-to-cart and subscription option, descriptions, cart and checkout button.  
   - **Not working:** Price not in a single clear element for the scraper (likely visible to users).  
   - **Takeaway:** Strong funnel; subscription (“Add Subscription to Cart”) is a clear differentiator.

2. **Ora Cacao**  
   - **Working:** Nothing (site timed out).  
   - **Not working:** Homepage did not reach `domcontentloaded` within 35s.  
   - **Takeaway:** Site performance or availability is a major risk for online sales; manual retest or different network may be needed.

3. **Four Visions (BigCommerce)**  
   - **Working:** Shop discoverable, ceremonial cacao category, add-to-cart, cart reachable, product story (e.g. Nana Marina’s Cacao), subscription options, PayPal messaging.  
   - **Not working:** Price in a complex block (script saw “price visible: false” despite $65 in page text); checkout CTA on cart not obvious to automation.  
   - **Takeaway:** Good product and trust (e.g. 30-day returns, secure payment); checkout discoverability from cart could be improved.

4. **Ma Cacao (Shopify)**  
   - **Working:** Shop discoverable, many product links (55), clear “Add to Cart”, descriptions, cart reachable (e.g. Ceremony Bar $12).  
   - **Not working:** Price again not in a single clear selector; checkout CTA on cart not obvious.  
   - **Takeaway:** Strong catalog and product page; ensuring a single, prominent “Checkout” on cart would align with best practice.

5. **Bar and Cocoa (Shopify)**  
   - **Working:** Shop discoverable, large catalog (64 product links), add-to-cart, descriptions, cart and checkout path clear.  
   - **Not working:** Price not in a single clear element (script); one product sampled was “Sold out” (inventory clarity is good).  
   - **Takeaway:** Best full funnel in this set: shop → product → cart → checkout all clearly reachable; multi-brand retailer execution is strong.

### Recommendations for Agroverse (from this research)

- **Ensure a single, prominent price element** on product pages (and cart) so both users and tools see price at a glance.
- **Make the checkout CTA on the cart page obvious** (one clear “Checkout” or “Proceed to checkout” button/link) to match Bar and Cocoa / Ceremonial Cacao.
- **Monitor site load time**; Ora Cacao’s timeout is a caution for dependency on slow or fragile hosting.
- **Consider subscription and one-click options** where they fit the brand; Ceremonial Cacao and Four Visions use subscriptions and PayPal/smart buttons.
